# LOCOS oxidation example configuration.
# Lengths are in micrometers, time is in hours, pressure is in atm.
#
# Geometry:
#   Si substrate + pad oxide (padOxideThickness) + Si3N4 mask box.
#   The mask covers x < maskEdge (default 0); the open window is x > maskEdge.
#   Bird's-beak oxide grows laterally under the mask edges.
#
# Grid sizing guide:
#   gridDelta=0.025 -> ~100k nodes (high-quality bird's beak)
#   gridDelta=0.05  ->  ~25k nodes (default, good quality)
#   gridDelta=0.1   ->   ~6k nodes (fast test)

numThreads=16
useGpu=gpu
gpuPreconditioner=jacobi

gridDelta=0.005
xExtent=1.
yMin=-1.
yMax=2.

padOxideThickness=0.03
maskThickness=0.05
maskEdge=0.

oxidationTime=0.1
# Output interval and maximum internal oxidation step. The model automatically
# subcycles below this value when the CFL limit is smaller.
timeStep=0.01
temperature=1000.
pressure=1.
oxidant=wet
orientation=100

# maxGridPoints=5000000
outputPrefix=ps_locos

pressureIterations=500
pressureTolerance=1e-3
stokesIterations=500
stokesTolerance=1e-3
mechanicsIterations=300
mechanicsTolerance=1e-2
couplingIterations=30
couplingTolerance=5e-2
maskCouplingIterations=30
maskCouplingTolerance=4e-2

# Log verbosity: error | warning | info (default) | timing | intermediate | debug
# Set to "timing" to print per-phase wall-time breakdowns alongside normal output.
logLevel=info
